#### 双指针前移法
```
#include <iostream>
#include <vector>
#include <stack>
using namespace std;

const int N = 1e5 + 5;
int n;

int partition(vector<int>& v, int left, int right) {
	int pre = left;
	int cur = pre + 1;
	int key = v[left];
	while (cur <= right) {
		if (v[cur] < key && ++pre != cur) swap(v[cur], v[pre]);
		++cur;
	}
	swap(v[left], v[pre]); // 将基准元素放回正确的位置
	return pre;
}

void quickSort(vector<int>& v, int left, int right) {
	stack<int> s;
	s.push(left);
	s.push(right);
	while (!s.empty()) {
		int r = s.top();
		s.pop();
		int l = s.top();
		s.pop();
		if (l >= r) continue;
		int boundary = partition(v, l, r);
		s.push(l);
		s.push(boundary - 1);
		s.push(boundary + 1);
		s.push(r);
	}
} 

int main() {
	cin >> n;
	vector<int> v;	
	int x;
	for (int i = 1; i <= n; ++i) {
		cin >> x;
		v.push_back(x);
	}	
	quickSort(v, 0, n - 1);
	for (auto element : v) {
		cout << element << " ";
	}
	return 0;
}
```
